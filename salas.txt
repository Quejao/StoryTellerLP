ruaIni -> (palacio, floresta)
	palacio -> (hall, patio, rua)
		hall -> (palacio, torre de vigia, sala do trono)
			sala do trono -> (hall)
		patio -> (palacio, torre de vigia)
			torre de vigia -> (hall, patio)
	floresta -> (atacar em arco, atacar com pulo, estocada)
		atacar com pulo -> (morte1)
		estocada -> (morte2)
		atacar em arco -> (floresta2)
			floresta2 -> (estradaPQ, mato a dentroEC, investigar rastros)
				estradaPQ -> (investigar, olhar penhasco, floresta2)
					olhar penhasco -> (morte3)
				mato a dentroEC -> (caverna, floresta2)
					caverna -> (esq, dir)
						esq -> (atacar, espreitar, caverna)
							atacar -> (morte4)
							espreitar -> (morte5)
						dir -> (golpe por cima, caverna)
							golpe por cima -> (vitoria)
							
rua = ruaIni
